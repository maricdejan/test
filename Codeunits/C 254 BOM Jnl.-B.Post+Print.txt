OBJECT Codeunit 254 BOM Jnl.-B.Post+Print
{
  OBJECT-PROPERTIES
  {
    Date=08.06.01;
    Time=12:00:00;
    Version List=NAVW13.00;
  }
  PROPERTIES
  {
    TableNo=234;
    OnRun=BEGIN
            BOMJnlBatch.COPY(Rec);
            Code;
            Rec := BOMJnlBatch;
          END;

  }
  CODE
  {
    VAR
      Text000@1000 : TextConst 'DEU=Mchten Sie die Buch.-Bltter buchen und den Buchungsbericht drucken?;ENU=Do you want to post the journals and print the posting report?';
      Text001@1001 : TextConst 'DEU=Die Buch.-Bltter sind gebucht worden.;ENU=The journals were successfully posted.';
      Text002@1002 : TextConst 'DEU="Es konnten nicht alle Buch.-Bltter gebucht werden. ";ENU="It was not possible to post all of the journals. "';
      Text003@1003 : TextConst 'DEU=Die nicht erfolgreich gebuchten Buch.-Bltter sind jetzt markiert.;ENU=The journals that were not successfully posted are now marked.';
      BOMJnlTemplate@1004 : Record 88;
      BOMJnlBatch@1005 : Record 234;
      BOMJnlLine@1006 : Record 89;
      BOMReg@1007 : Record 239;
      BOMJnlPostBatch@1008 : Codeunit 33;
      JnlWithErrors@1009 : Boolean;

    LOCAL PROCEDURE Code@1();
    BEGIN
      WITH BOMJnlBatch DO BEGIN
        BOMJnlTemplate.GET("Journal Template Name");
        BOMJnlTemplate.TESTFIELD("Posting Report ID");

        IF NOT CONFIRM(Text000,FALSE) THEN
          EXIT;

        FIND('-');
        REPEAT
          BOMJnlLine."Journal Template Name" := "Journal Template Name";
          BOMJnlLine."Journal Batch Name" := Name;
          BOMJnlLine."Line No." := 1;
          CLEAR(BOMJnlPostBatch);
          IF BOMJnlPostBatch.RUN(BOMJnlLine) THEN BEGIN
            MARK(FALSE);
            IF BOMReg.GET(BOMJnlLine."Line No.") THEN BEGIN
              BOMReg.SETRECFILTER;
              REPORT.RUN(BOMJnlTemplate."Posting Report ID",FALSE,FALSE,BOMReg);
            END;
          END ELSE BEGIN
            MARK(TRUE);
            JnlWithErrors := TRUE;
          END;
        UNTIL NEXT = 0;

        IF NOT JnlWithErrors THEN
          MESSAGE(Text001)
        ELSE
          MESSAGE(
            Text002 +
            Text003);

        IF NOT FIND('=><') THEN BEGIN
          RESET;
          Name := '';
        END;
      END;
    END;

    BEGIN
    END.
  }
}
