    PROCEDURE CreateDiffListBuffer@1140000(PhysInvtOrderLine@1140000 : Record 5005351;VAR NoOfBufferLines@1140001 : Integer);
    VAR
      ExpPhysInvtTrackLine@1140002 : Record 5005361;
      PhysInvtRecordingLine@1140003 : Record 5005353;
      ReservEntry@1140004 : Record 337;
      NextLineNo@1140005 : Integer;
    BEGIN
      WITH PhysInvtOrderLine DO BEGIN
        NoOfBufferLines := 0;

        NextLineNo := 1;
        ExpPhysInvtTrackLine.RESET;
        ExpPhysInvtTrackLine.SETRANGE("Order No", "Document No.");
        ExpPhysInvtTrackLine.SETRANGE("Order Line No.", "Line No.");
        IF ExpPhysInvtTrackLine.FIND('-') THEN REPEAT
          FindOrCreateDiffListBuffer(NoOfBufferLines, NextLineNo);
          TempPhysInvtDiffListBuffer."Exp. Serial No." := ExpPhysInvtTrackLine."Serial No.";
          TempPhysInvtDiffListBuffer."Exp. Lot No." := ExpPhysInvtTrackLine."Lot No.";
          TempPhysInvtDiffListBuffer."Exp. Qty. (Base)" := ExpPhysInvtTrackLine."Quantity (Base)";
          TempPhysInvtDiffListBuffer.MODIFY;
        UNTIL ExpPhysInvtTrackLine.NEXT = 0;

        NextLineNo := 1;
        PhysInvtRecordingLine.RESET;
        PhysInvtRecordingLine.SETCURRENTKEY("Order No.", "Order Line No.");
        PhysInvtRecordingLine.SETRANGE("Order No.", "Document No.");
        PhysInvtRecordingLine.SETRANGE("Order Line No.", "Line No.");
        IF PhysInvtRecordingLine.FIND('-') THEN REPEAT
          FindOrCreateDiffListBuffer(NoOfBufferLines, NextLineNo);
          TempPhysInvtDiffListBuffer."Rec. No." := PhysInvtRecordingLine."Recording No.";
          TempPhysInvtDiffListBuffer."Rec. Line No." := PhysInvtRecordingLine."Line No.";
          TempPhysInvtDiffListBuffer."Rec. Serial No." := PhysInvtRecordingLine."Serial No.";
          TempPhysInvtDiffListBuffer."Rec. Lot No." := PhysInvtRecordingLine."Lot No.";
          TempPhysInvtDiffListBuffer."Rec. Qty. (Base)" := PhysInvtRecordingLine."Quantity (Base)";
          TempPhysInvtDiffListBuffer.MODIFY;
        UNTIL PhysInvtRecordingLine.NEXT = 0;

        NextLineNo := 1;
        ReservEntry.RESET;
        ReservEntry.SETCURRENTKEY(
          "Source Type", "Source Subtype", "Source ID",
          "Source Batch Name", "Source Prod. Order Line",
          "Source Ref. No.");
        ReservEntry.SETRANGE("Source Type", DATABASE::"Phys. Inventory Order Line");
        ReservEntry.SETRANGE("Source Subtype", 0);
        ReservEntry.SETRANGE("Source ID", "Document No.");
        ReservEntry.SETRANGE("Source Batch Name", '');
        ReservEntry.SETRANGE("Source Prod. Order Line", 0);
        ReservEntry.SETRANGE("Source Ref. No.", "Line No.");
        IF ReservEntry.FIND('-') THEN REPEAT
          FindOrCreateDiffListBuffer(NoOfBufferLines, NextLineNo);
          TempPhysInvtDiffListBuffer."Track. Serial No." := ReservEntry."Serial No.";
          TempPhysInvtDiffListBuffer."Track. Lot No." := ReservEntry."Lot No.";

            TempPhysInvtDiffListBuffer."Track. Qty. Neg. (Base)" := 0;
            TempPhysInvtDiffListBuffer."Track. Qty. Pos. (Base)" := ReservEntry.Quantity;
