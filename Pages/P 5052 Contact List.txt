    BEGIN
      IF Type = Type::Company THEN
        NameEmphasize := TRUE
      ELSE BEGIN
        Cont.SETCURRENTKEY("Company Name","Company No.",Type,Name);
        IF ("Company No." <> '') AND (NOT HASFILTER) AND (NOT MARKEDONLY) AND (CURRENTKEY = Cont.CURRENTKEY)

          NameIndent := 1;
