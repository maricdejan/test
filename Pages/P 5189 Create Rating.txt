    LOCAL PROCEDURE ShowStep@2(Visible@1001 : Boolean);
    BEGIN
      CASE "Wizard Step" OF
        "Wizard Step"::"1":
          BEGIN
            NextEnable := TRUE;
            BackEnable := FALSE;
            Step1Visible := Visible;
            IF Visible THEN;
          END;
        "Wizard Step"::"2":
          BEGIN
            Step2Visible := Visible;
            BackEnable := TRUE;
            NextEnable := TRUE;
          END;
        "Wizard Step"::"3":
          BEGIN
            Step3Visible := Visible;
            IF Visible THEN BEGIN
              BackEnable := TRUE;
              NextEnable := TRUE;
              FinishEnable := FALSE;
            END;
          END;
        "Wizard Step"::"4":
          BEGIN
            IF Visible THEN BEGIN
              GetAnswers(TempProfileLineAnswer);

          END;
      END;
    END;
